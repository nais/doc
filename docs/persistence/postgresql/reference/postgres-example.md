This is a complete example of a `Postgres` resource, commonly known as the `postgres.yaml` file.

For an in-depth explanation of each field, head over to the [reference documentation](./postgres-spec.md).

``` yaml
apiVersion: data.nais.io/v1
kind: Postgres
metadata:
  creationTimestamp: null
  labels:
    team: myteam
  name: mycluster
  namespace: myteam
spec:
  cluster:
    allowDeletion: true
    audit:
      enabled: true
      statementClasses:
      - function
      - misc
    highAvailability: true
    majorVersion: "17"
    resources:
      cpu: 200m
      diskSize: 2Gi
      memory: 2Gi
  database:
    collation: nb_NO.UTF-8
    extensions:
    - name: postgis
  maintenanceWindow:
    day: 4
    hour: 10
```
